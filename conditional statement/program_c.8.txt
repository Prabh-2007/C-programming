#include <stdio.h>
int main(){
    int a,b,ch;
    printf("enter choice(1-add,2-sub,3-mul,4-div:");
    scanf("%d",&ch);
    printf("enter two numbers:");
    scanf("%d %d",&a,&b);
    if(ch==1){
        printf("result = %d",a+b);
    }
    else if(ch==2){
        printf("result = %d",a-b);
    }
    else if(ch == 3){
        printf("result = %d",a*b);
    }
    else if(ch == 4){
        printf("result = %d",a/b);
    }
    else{
        printf("invalid value");
    }
    return 0;
}
