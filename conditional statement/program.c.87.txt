#include <stdio.h>
int main() {
    int x, y;
    scanf("%d", &x);
    scanf("%d", &y);
    if(x > 0 && y > 0)
        printf("1st Quadrant");
    else if(x < 0 && y > 0)
        printf("2nd Quadrant");
    else if(x < 0 && y < 0)
        printf("3rd Quadrant");
    else if(x > 0 && y < 0)
        printf("4th Quadrant");
    else if(x == 0 && y == 0)
        printf("Origin");
    else if(x == 0)
        printf("On Y-axis");
    else
        printf("On X-axis");

    return 0;
}
