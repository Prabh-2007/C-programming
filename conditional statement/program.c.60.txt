#include <stdio.h>

int main() {

    int day, month, year;
    int quarter, days = 0;

    printf("Enter date (DD MM YYYY): ");
    scanf("%d %d %d", &day, &month, &year);

    // Basic month validation
    if(month < 1 || month > 12) {
        printf("Invalid Month");
        return 0;
    }

    // -------- FIND QUARTER --------
    if(month >= 1 && month <= 3)
        quarter = 1;
    else if(month >= 4 && month <= 6)
        quarter = 2;
    else if(month >= 7 && month <= 9)
        quarter = 3;
    else
        quarter = 4;

    // -------- CALCULATE DAYS PASSED IN QUARTER --------
    
    if(quarter == 1) {   // Jan, Feb, Mar
        if(month == 1)
            days = day;
        else if(month == 2)
            days = 31 + day;
        else
            days = 31 + 28 + day;   // Feb = 28 (simple version)
    }

    else if(quarter == 2) {   // Apr, May, Jun
        if(month == 4)
            days = day;
        else if(month == 5)
            days = 30 + day;
        else
            days = 30 + 31 + day;
    }

    else if(quarter == 3) {   // Jul, Aug, Sep
        if(month == 7)
            days = day;
        else if(month == 8)
            days = 31 + day;
        else
            days = 31 + 31 + day;
    }

    else {   // Quarter 4 â†’ Oct, Nov, Dec
        if(month == 10)
            days = day;
        else if(month == 11)
            days = 31 + day;
        else
            days = 31 + 30 + day;
    }

    // -------- FIND SEASON --------

    if((month == 3) || (month == 4) || (month == 5))
        printf("Season: Spring\n");
    else if((month == 6) || (month == 7) || (month == 8))
        printf("Season: Summer\n");
    else if((month == 9) || (month == 10) || (month == 11))
        printf("Season: Autumn\n");
    else
        printf("Season: Winter\n");

    printf("Quarter: Q%d\n", quarter);
    printf("Days passed in Quarter: %d\n", days);

    return 0;
}
