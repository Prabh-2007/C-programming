#include <stdio.h>

int main() {

    char a,b;
    int x,y;

    scanf(" %c %c",&a,&b);

    x = a;
    y = b;

    int v1 = (a=='a'||a=='e'||a=='i'||a=='o'||a=='u');
    int v2 = (b=='a'||b=='e'||b=='i'||b=='o'||b=='u');

    if(v1 && v2)
        printf("%d", x+y);

    else if(v1 || v2)
        printf("%d", x-y);

    else
        printf("%d", x*y);

    return 0;
}
