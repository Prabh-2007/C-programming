#include <stdio.h>

int main()
{
    int n,r,flag=1;

    scanf("%d",&n);

    while(n>0)
    {
        r=n%10;

        if(r!=0 && r!=1)
            flag=0;

        n=n/10;
    }

    if(flag==1)
        printf("Binary Number");
    else
        printf("Not Binary");

    return 0;
}
