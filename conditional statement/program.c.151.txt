#include <stdio.h>
#include <math.h>

int main()
{
    int n,temp,count=0,sum=0,r,i=0;
    scanf("%d",&n);

    temp=n;

    while(temp>0)
    {
        count++;
        temp=temp/10;
    }

    temp=n;

    while(temp>0)
    {
        r=temp%10;
        sum=sum+pow(r,count);
        count--;
        temp=temp/10;
    }

    if(sum==n)
        printf("Disarium Number");
    else
        printf("Not Disarium");

    return 0;
}
