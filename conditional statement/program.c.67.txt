#include<stdio.h>
int main(){
    int num ,original,remainder,reverse = 0;
    scanf("%d",&num);
    original = num;
    while(num != 0){
        remainder = num%10;
        reverse = reverse*10 + remainder;
        num = num/10;
    }
    if(original == reverse){
        printf("yes this is palindrome");
    }
    else{
        printf("it's not palindrome");
    }
    return 0;
}
