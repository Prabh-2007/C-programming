#include <stdio.h>

int main()
{
    float x1, y1, x2, y2, slope;

    printf("Enter x1: ");
    scanf("%f", &x1);

    printf("Enter y1: ");
    scanf("%f", &y1);

    printf("Enter x2: ");
    scanf("%f", &x2);

    printf("Enter y2: ");
    scanf("%f", &y2);

    if(x2 - x1 == 0)
    {
        printf("Slope is undefined");
    }
    else
    {
        slope = (y2 - y1) / (x2 - x1);

        printf("Slope = %.2f\n", slope);

        if(slope > 0)
        {
            printf("The land surface rises.");
        }
        else if(slope < 0)
        {
            printf("The land surface descends.");
        }
        else
        {
            printf("The land surface is horizontal.");
        }
    }

    return 0;
}
