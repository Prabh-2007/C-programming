#include <stdio.h>
int main()
{
    int n, sum=0, r;

    scanf("%d",&n);

    while(n>9)
    {
        sum=0;

        while(n>0)
        {
            r=n%10;
            sum=sum+r;
            n=n/10;
        }

        n=sum;
    }

    if(n==1)
        printf("Magic Number");
    else
        printf("Not Magic");

    return 0;
}
