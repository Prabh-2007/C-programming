#include <stdio.h>

int main() {
    int n1, n2, n3, n4, n5;

    scanf("%d %d %d %d %d", &n1, &n2, &n3, &n4, &n5);

    if(n1 >= 0 && n1 <= 9)
        printf("%d", n1);
    else
        printf("Invalid");

    if(n2 >= 0 && n2 <= 9)
        printf("%d", n2);
    else
        printf("Invalid");

    if(n3 >= 0 && n3 <= 9)
        printf("%d", n3);
    else
        printf("Invalid");

    if(n4 >= 0 && n4 <= 9)
        printf("%d", n4);
    else
        printf("Invalid");

    if(n5 >= 0 && n5 <= 9)
        printf("%d", n5);
    else
        printf("Invalid");

    return 0;
}
