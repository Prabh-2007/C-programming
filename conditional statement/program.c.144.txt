#include <stdio.h>
int main()
{
    int n, rev=0, r;

    scanf("%d",&n);

    if(n<0)
        printf("Negative Number");
    else
    {
        while(n>0)
        {
            r=n%10;
            rev=rev*10+r;
            n=n/10;
        }

        printf("Reverse = %d",rev);
    }

    return 0;
}
