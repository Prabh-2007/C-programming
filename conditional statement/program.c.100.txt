#include <stdio.h>

int main() {
    int temp;

    printf("Enter temperature: ");
    scanf("%d", &temp);

    if(temp < 0)
        printf("Freezing");
    else if(temp <= 25)
        printf("Cold");
    else if(temp <= 35)
        printf("Warm");
    else
        printf("Hot");

    return 0;
}
