#include <stdio.h>

int main() {
    int num, i, sum = 0;
    scanf("%d", &num);
    for(i = 1; i < num; i++) {
        if(num % i == 0) {
            sum = sum + i;
        }
    }
    if(sum == num)
        printf("It is a Perfect Number");
    else
        printf("It is NOT a Perfect Number");
    return 0;
}
