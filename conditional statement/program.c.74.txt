#include <stdio.h>
#include <math.h>

int main() {
    int hour, minute;
    float hourAngle, minuteAngle, angle;

    printf("Enter hour (0-23): ");
    scanf("%d", &hour);

    printf("Enter minute (0-59): ");
    scanf("%d", &minute);

    // Convert hour to 12-hour format
    hour = hour % 12;

    minuteAngle = minute * 6;
    hourAngle = (hour * 30) + (minute * 0.5);

    angle = fabs(hourAngle - minuteAngle);

    // Optional: smallest angle
    if (angle > 180)
        angle = 360 - angle;

    printf("Angle between hands = %.2f degrees", angle);

    return 0;
}
